    <script>
      const audio = document.getElementById('applause');
      const fallback = document.getElementById('autoplay-fallback');
      const playBtn = document.getElementById('playBtn');

      // Try to autoplay (muted attempt first to increase chance it plays automatically)
      (async () => {
        try {
          audio.muted = true;
          await audio.play();
          // briefly unmute (some browsers allow muted autoplay but not unmute programmatically;
          // leaving it muted might be undesirable, so we attempt to unmute)
          setTimeout(() => { audio.muted = false; }, 400);
        } catch (err) {
          // autoplay blocked ‚Äî show fallback button
          audio.pause();
          audio.muted = false;
          fallback.style.display = 'block';
        }
      })();

      playBtn.addEventListener('click', async () => {
        try {
          await audio.play();
          fallback.style.display = 'none';
        } catch (e) {
          console.error('Play failed:', e);
        }
      });
    </script>
